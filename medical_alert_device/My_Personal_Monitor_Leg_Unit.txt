/*
  Personal Monitor - Leg Unit

  This sketch reads motion data from an MPU6050-class accelerometer/gyroscope,
  estimates body position, reads a local body-temperature sensor, and transmits
  the results to the display unit over ESP-NOW.
*/

#include <Wire.h>
#include <WiFi.h>
#include <esp_now.h>

// -----------------------------------------------------------------------------
// Sensor configuration
// -----------------------------------------------------------------------------
constexpr int MPU_ADDRESS = 0x68;
constexpr int BODY_TEMP_SENSOR_PIN = 33;

// Update this MAC address to match the display unit.
uint8_t broadcastAddress[] = {0xCC, 0xDB, 0xA7, 0x64, 0x33, 0x60};

// -----------------------------------------------------------------------------
// Motion sensor state
// -----------------------------------------------------------------------------
int16_t accelX = 0;
int16_t accelY = 0;
int16_t accelZ = 0;
int16_t gyroX = 0;
int16_t gyroY = 0;
int16_t gyroZ = 0;

unsigned long prevWalkTime = 0;

// -----------------------------------------------------------------------------
// Data payload
// -----------------------------------------------------------------------------
constexpr int IDENTIFIER_NUMBER = 4;
int positionCode = 0;
int bodyTempF = 0;

typedef struct struct_message {
  int a;
  char b[32];
  float c;
  int d;
} struct_message;

struct_message myData;
esp_now_peer_info_t peerInfo;

// -----------------------------------------------------------------------------
// Function declarations
// -----------------------------------------------------------------------------
void accelSetup();
void getAccelData();
void getBodyPosition();
void bodyTempSensor();
void sendData();

// -----------------------------------------------------------------------------
// ESP-NOW send callback
// -----------------------------------------------------------------------------
void OnDataSent(const uint8_t* mac_addr, esp_now_send_status_t status) {
  Serial.println(status == ESP_NOW_SEND_SUCCESS ? "Delivery success" : "Delivery failed");
}

void accelSetup() {
  Wire.begin();
  Wire.beginTransmission(MPU_ADDRESS);
  Wire.write(0x6B);
  Wire.write(0);
  Wire.endTransmission(true);
}

void getAccelData() {
  Wire.beginTransmission(MPU_ADDRESS);
  Wire.write(0x3B);
  Wire.endTransmission(false);
  Wire.requestFrom(MPU_ADDRESS, 12, true);

  accelX = Wire.read() << 8 | Wire.read();
  accelY = Wire.read() << 8 | Wire.read();
  accelZ = Wire.read() << 8 | Wire.read();
  gyroX = Wire.read() << 8 | Wire.read();
  gyroY = Wire.read() << 8 | Wire.read();
  gyroZ = Wire.read() << 8 | Wire.read();

  Serial.print("Accelerometer: X=");
  Serial.print(accelX);
  Serial.print(" | Y=");
  Serial.print(accelY);
  Serial.print(" | Z=");
  Serial.println(accelZ);

  Serial.print("Gyroscope: X=");
  Serial.print(gyroX);
  Serial.print(" | Y=");
  Serial.print(gyroY);
  Serial.print(" | Z=");
  Serial.println(gyroZ);
  Serial.println();

  delay(100);
}

void getBodyPosition() {
  // These thresholds are based on the current sensor mounting orientation.
  if (accelY < -10000 && gyroZ > 6000) {
    positionCode = 400;  // Walking
  } else if (accelY < -10000 && (millis() - prevWalkTime > 4000)) {
    positionCode = 100;  // Standing
  } else if (accelY > -3000) {
    positionCode = 200;  // Sitting
  } else {
    positionCode = 0;
  }

  switch (positionCode) {
    case 100:
      strncpy(myData.b, "Standing", sizeof(myData.b) - 1);
      myData.b[sizeof(myData.b) - 1] = '\0';
      break;

    case 200:
      strncpy(myData.b, "Sitting", sizeof(myData.b) - 1);
      myData.b[sizeof(myData.b) - 1] = '\0';
      break;

    case 300:
      strncpy(myData.b, "Lying Down", sizeof(myData.b) - 1);
      myData.b[sizeof(myData.b) - 1] = '\0';
      break;

    case 400:
      strncpy(myData.b, "Walking", sizeof(myData.b) - 1);
      myData.b[sizeof(myData.b) - 1] = '\0';
      prevWalkTime = millis();
      break;

    case 500:
      strncpy(myData.b, "Running", sizeof(myData.b) - 1);
      myData.b[sizeof(myData.b) - 1] = '\0';
      break;

    default:
      strncpy(myData.b, "Unknown", sizeof(myData.b) - 1);
      myData.b[sizeof(myData.b) - 1] = '\0';
      break;
  }

  positionCode = 0;
}

void bodyTempSensor() {
  int sensorValue = analogRead(BODY_TEMP_SENSOR_PIN);
  float voltage = sensorValue * (3.3f / 4095.0f);
  int temperatureC = static_cast<int>((voltage - 0.5f) * 100.0f);

  // The Fahrenheit offset reflects the current sensor calibration used in the
  // original project.
  bodyTempF = static_cast<int>((temperatureC * 1.8f) + 62.0f);

  Serial.print(bodyTempF);
  Serial.println(": Body temperature");
}

void sendData() {
  myData.a = IDENTIFIER_NUMBER;
  myData.d = bodyTempF;

  Serial.print("myData.d: ");
  Serial.println(myData.d);

  esp_err_t result = esp_now_send(broadcastAddress, (uint8_t*)&myData, sizeof(myData));

  if (result == ESP_OK) {
    Serial.println("Transmission confirmed");
  } else {
    Serial.println("Transmission error");
  }
}

void setup() {
  Serial.begin(115200);
  pinMode(BODY_TEMP_SENSOR_PIN, INPUT);

  accelSetup();

  WiFi.mode(WIFI_STA);

  if (esp_now_init() != ESP_OK) {
    Serial.println("Error initializing ESP-NOW");
    return;
  }

  esp_now_register_send_cb(OnDataSent);

  memcpy(peerInfo.peer_addr, broadcastAddress, 6);
  peerInfo.channel = 0;
  peerInfo.encrypt = false;

  if (esp_now_add_peer(&peerInfo) != ESP_OK) {
    Serial.println("Failed to add peer");
    return;
  }
}

void loop() {
  getAccelData();
  getBodyPosition();
  bodyTempSensor();
  sendData();

  delay(250);
}
