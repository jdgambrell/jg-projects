/*
  Personal Monitor - Wrist Unit

  This sketch reads pulse data from a MAX30102/MAX30105-based sensor and sends
  heart-rate information to the display unit over ESP-NOW.
*/

#include <Wire.h>
#include <WiFi.h>
#include <esp_now.h>
#include "MAX30105.h"
#include "heartRate.h"

MAX30105 particleSensor;

// -----------------------------------------------------------------------------
// Heart-rate sampling configuration
// -----------------------------------------------------------------------------
constexpr byte RATE_SIZE = 4;
byte rates[RATE_SIZE] = {0};
byte rateSpot = 0;
long lastBeat = 0;

float beatsPerMinute = 0.0f;
int beatAvg = 0;

// -----------------------------------------------------------------------------
// Data payload and transmit timing
// -----------------------------------------------------------------------------
constexpr int IDENTIFIER_NUMBER = 2;
unsigned long prevSendTime = 0;
constexpr unsigned long SEND_INTERVAL_MS = 250;

// Update this MAC address to match the display unit.
uint8_t broadcastAddress[] = {0xCC, 0xDB, 0xA7, 0x64, 0x33, 0x60};

typedef struct struct_message {
  int a;
  char b[32];
  float c;
  int d;
} struct_message;

struct_message myData;
esp_now_peer_info_t peerInfo;

// -----------------------------------------------------------------------------
// Function declarations
// -----------------------------------------------------------------------------
void initializeMAX30102();
void getBPM();
void sendData();

// -----------------------------------------------------------------------------
// ESP-NOW send callback
// -----------------------------------------------------------------------------
void OnDataSent(const uint8_t* mac_addr, esp_now_send_status_t status) {
  Serial.print("\r\nLast packet send status: ");
  Serial.println(status == ESP_NOW_SEND_SUCCESS ? "Delivery success" : "Delivery failed");
}

void initializeMAX30102() {
  Serial.println("Initializing sensor...");

  if (!particleSensor.begin(Wire, I2C_SPEED_FAST)) {
    Serial.println("MAX30105 was not found. Verify wiring and power.");
    while (1) {
      // Halt until the hardware issue is corrected.
    }
  }

  Serial.println("Place your index finger on the sensor with steady pressure.");

  particleSensor.setup();
  particleSensor.setPulseAmplitudeRed(0x0A);
  particleSensor.setPulseAmplitudeGreen(0);
}

void getBPM() {
  long irValue = particleSensor.getIR();

  if (checkForBeat(irValue)) {
    long delta = millis() - lastBeat;
    lastBeat = millis();

    beatsPerMinute = 60.0f / (delta / 1000.0f);

    if (beatsPerMinute < 255 && beatsPerMinute > 20) {
      rates[rateSpot++] = static_cast<byte>(beatsPerMinute);
      rateSpot %= RATE_SIZE;

      beatAvg = 0;
      for (byte i = 0; i < RATE_SIZE; i++) {
        beatAvg += rates[i];
      }
      beatAvg /= RATE_SIZE;
    }
  }

  Serial.print("IR=");
  Serial.print(irValue);
  Serial.print(", BPM=");
  Serial.print(beatsPerMinute);
  Serial.print(", Avg BPM=");
  Serial.print(beatAvg);

  if (irValue < 50000) {
    Serial.print(" No finger detected");
    beatAvg = -1;
  }

  Serial.println();
}

void sendData() {
  myData.a = IDENTIFIER_NUMBER;
  myData.c = beatsPerMinute;
  myData.d = beatAvg;

  esp_err_t result = esp_now_send(broadcastAddress, (uint8_t*)&myData, sizeof(myData));

  if (result == ESP_OK) {
    Serial.println("Transmission confirmed");
  } else {
    Serial.println("Transmission error");
  }
}

void setup() {
  Serial.begin(115200);
  initializeMAX30102();

  WiFi.mode(WIFI_STA);

  if (esp_now_init() != ESP_OK) {
    Serial.println("Error initializing ESP-NOW");
    return;
  }

  esp_now_register_send_cb(OnDataSent);

  memcpy(peerInfo.peer_addr, broadcastAddress, 6);
  peerInfo.channel = 0;
  peerInfo.encrypt = false;

  if (esp_now_add_peer(&peerInfo) != ESP_OK) {
    Serial.println("Failed to add peer");
    return;
  }
}

void loop() {
  getBPM();

  if (millis() - prevSendTime > SEND_INTERVAL_MS) {
    sendData();
    prevSendTime = millis();
  }
}
