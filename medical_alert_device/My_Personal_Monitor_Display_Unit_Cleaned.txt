/*
  Personal Monitor - Display Unit

  This sketch receives health and posture data from the wrist and leg ESP32
  units over ESP-NOW, reads a local room-temperature sensor, updates an OLED
  display, and provides keypad-triggered warning and emergency alert modes.
*/

#include <Wire.h>
#include <SPI.h>
#include <WiFi.h>
#include <esp_now.h>
#include <Key.h>
#include <Keypad.h>
#include <Keypad_I2C.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

// -----------------------------------------------------------------------------
// Pin assignments
// -----------------------------------------------------------------------------
constexpr int ROOM_TEMP_PIN = 33;
constexpr int EMERGENCY_ALERT_PIN = 32;
constexpr int WARNING_ALERT_PIN = 18;
constexpr int NORMAL_OPERATION_PIN = 19;

// -----------------------------------------------------------------------------
// PWM configuration for the alert output
// -----------------------------------------------------------------------------
constexpr int PWM_FREQUENCY = 1000;
constexpr int PWM_CHANNEL = 0;
constexpr int PWM_RESOLUTION = 8;

// -----------------------------------------------------------------------------
// Keypad configuration
// -----------------------------------------------------------------------------
constexpr uint8_t I2C_ADDRESS = 0x38;
constexpr byte ROWS = 4;
constexpr byte COLS = 4;

char keypadLayout[ROWS][COLS] = {
  {'1', '2', '3', 'A'},
  {'4', '5', '6', 'B'},
  {'7', '8', '9', 'C'},
  {'*', '0', '#', 'D'}
};

byte rowPins[ROWS] = {4, 5, 6, 7};
byte colPins[COLS] = {0, 1, 2, 3};

Keypad_I2C keypad(makeKeymap(keypadLayout), rowPins, colPins, ROWS, COLS, I2C_ADDRESS, PCF8574);
char buttonPress = '\0';

// -----------------------------------------------------------------------------
// OLED configuration
// -----------------------------------------------------------------------------
constexpr int SCREEN_WIDTH = 128;
constexpr int SCREEN_HEIGHT = 64;
constexpr int OLED_RESET = -1;
constexpr int SCREEN_ADDRESS = 0x3C;

Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

// -----------------------------------------------------------------------------
// ESP-NOW message structure
// -----------------------------------------------------------------------------
typedef struct struct_message {
  int a;
  char b[32];
  float c;
  int d;
} struct_message;

struct_message myData;

// -----------------------------------------------------------------------------
// Sensor and display state
// -----------------------------------------------------------------------------
int roomTempFahr = 0;
int roomTempCelsius = 0;
unsigned long prevRoomTempTime = 0;
unsigned long prevOLEDTime = 0;

char bodyPosition[32] = "Unknown";
float beatsPerMinute = 0.0f;
int avgBeatsPerMinute = 0;
int bodyTemp = 0;

// -----------------------------------------------------------------------------
// Function declarations
// -----------------------------------------------------------------------------
void keyPadSetup();
void readKeys();
void readRoomTempSensor();
void dataToOLED();
void alertMonitor();
void warningAlert();
void emergencyAlert();
void showCenteredScreen(const String& line1, const String& line2 = "", const String& line3 = "");

// -----------------------------------------------------------------------------
// ESP-NOW receive callback
// -----------------------------------------------------------------------------
void OnDataRecv(const uint8_t* mac, const uint8_t* incomingData, int len) {
  memcpy(&myData, incomingData, sizeof(myData));

  Serial.print("Data received: ");
  Serial.println(len);
  Serial.print("Identifier: ");
  Serial.println(myData.a);
  Serial.print("Text value: ");
  Serial.println(myData.b);
  Serial.print("Float value: ");
  Serial.println(myData.c);
  Serial.print("Integer value: ");
  Serial.println(myData.d);
  Serial.println();

  if (myData.a == 4) {
    strncpy(bodyPosition, myData.b, sizeof(bodyPosition) - 1);
    bodyPosition[sizeof(bodyPosition) - 1] = '\0';
    bodyTemp = myData.d;
  }

  if (myData.a == 2) {
    beatsPerMinute = myData.c;
    avgBeatsPerMinute = myData.d;
  }
}

void setup() {
  Serial.begin(115200);
  Wire.begin();
  keyPadSetup();

  pinMode(ROOM_TEMP_PIN, INPUT);
  pinMode(EMERGENCY_ALERT_PIN, OUTPUT);
  pinMode(WARNING_ALERT_PIN, OUTPUT);
  pinMode(NORMAL_OPERATION_PIN, OUTPUT);

  digitalWrite(EMERGENCY_ALERT_PIN, LOW);
  digitalWrite(WARNING_ALERT_PIN, LOW);
  digitalWrite(NORMAL_OPERATION_PIN, HIGH);

  ledcSetup(PWM_CHANNEL, PWM_FREQUENCY, PWM_RESOLUTION);
  ledcAttachPin(EMERGENCY_ALERT_PIN, PWM_CHANNEL);

  WiFi.mode(WIFI_STA);

  if (esp_now_init() != ESP_OK) {
    Serial.println("Error initializing ESP-NOW");
    return;
  }

  esp_now_register_recv_cb(OnDataRecv);

  if (!display.begin(SSD1306_SWITCHCAPVCC, SCREEN_ADDRESS)) {
    Serial.println(F("SSD1306 allocation failed"));
    for (;;) {
      // Halt if the display cannot be initialized.
    }
  }

  display.clearDisplay();
  display.display();
}

void loop() {
  readRoomTempSensor();
  dataToOLED();
  readKeys();
  alertMonitor();
}

void keyPadSetup() {
  Serial.println("Initializing keypad expander");
  keypad.begin(makeKeymap(keypadLayout));
}

void readKeys() {
  buttonPress = keypad.getKey();

  if (buttonPress) {
    Serial.print("Key pressed: ");
    Serial.println(buttonPress);
  }
}

void showCenteredScreen(const String& line1, const String& line2, const String& line3) {
  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE);
  display.setTextSize(2);
  display.setCursor(0, 0);
  display.println(line1);

  if (line2.length() > 0) {
    display.println(line2);
  }

  if (line3.length() > 0) {
    display.println(line3);
  }

  display.display();
}

void dataToOLED() {
  unsigned long elapsed = millis() - prevOLEDTime;

  if (elapsed < 11000) {
    display.clearDisplay();
    display.setTextColor(SSD1306_WHITE);
    display.setTextSize(3);
    display.setCursor(0, 0);
    display.println(bodyPosition);
    display.setTextSize(2);
    display.setCursor(0, 35);
    display.print("Position");
    display.display();
  } else if (elapsed <= 18000) {
    display.clearDisplay();
    display.setTextColor(SSD1306_WHITE);
    display.setTextSize(3);
    display.setCursor(35, 0);
    display.println(avgBeatsPerMinute);
    display.setTextSize(2);
    display.println();
    display.print("BPM");
    display.display();
  } else if (elapsed <= 23000) {
    display.clearDisplay();
    display.setTextColor(SSD1306_WHITE);
    display.setTextSize(3);
    display.setCursor(0, 0);
    display.print(bodyTemp);
    display.println(" F");
    display.setTextSize(2);
    display.setCursor(0, 35);
    display.print("Body Temp.");
    display.display();
  } else if (elapsed <= 27000) {
    display.clearDisplay();
    display.setTextColor(SSD1306_WHITE);
    display.setTextSize(3);
    display.setCursor(0, 0);
    display.print(roomTempFahr);
    display.println(" F");
    display.setTextSize(2);
    display.setCursor(0, 35);
    display.print("Room Temp.");
    display.display();
  } else {
    prevOLEDTime = millis();
  }
}

void readRoomTempSensor() {
  if (millis() - prevRoomTempTime <= 2000) {
    return;
  }

  int sensorValue = analogRead(ROOM_TEMP_PIN);
  float voltage = sensorValue * (3.3f / 4095.0f);

  roomTempCelsius = static_cast<int>((voltage - 0.5f) / 0.01f);
  roomTempFahr = static_cast<int>((roomTempCelsius * (9.0f / 5.0f)) + 62.0f);

  Serial.print("Room temperature: ");
  Serial.println(roomTempFahr);

  prevRoomTempTime = millis();
}

void emergencyAlert() {
  digitalWrite(EMERGENCY_ALERT_PIN, HIGH);
  digitalWrite(NORMAL_OPERATION_PIN, LOW);

  showCenteredScreen("Simulates", "critical HR", "and temp");
  delay(5000);

  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE);
  display.setTextSize(2);
  display.setCursor(0, 0);
  display.println("Emergency");
  display.setCursor(0, 25);
  display.println("Patient");
  display.print("Flatlining");
  display.display();

  unsigned long startTime = millis();
  while (millis() - startTime < 10000) {
    for (int dutyCycle = 0; dutyCycle <= 255; dutyCycle++) {
      ledcWrite(PWM_CHANNEL, dutyCycle);
      delay(5);
    }
  }

  ledcWrite(PWM_CHANNEL, 0);
  digitalWrite(EMERGENCY_ALERT_PIN, LOW);
  digitalWrite(NORMAL_OPERATION_PIN, HIGH);
  buttonPress = '0';
}

void warningAlert() {
  digitalWrite(WARNING_ALERT_PIN, HIGH);
  digitalWrite(NORMAL_OPERATION_PIN, LOW);

  showCenteredScreen("Simulates", "high HR", "and temp");
  delay(6000);

  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE);
  display.setTextSize(2);
  display.setCursor(0, 0);
  display.println("Caution");
  display.setCursor(0, 25);
  display.println("Overheat");
  display.print("Warning");
  display.display();

  delay(10000);
  digitalWrite(WARNING_ALERT_PIN, LOW);
  digitalWrite(NORMAL_OPERATION_PIN, HIGH);
  buttonPress = '0';
}

void alertMonitor() {
  if (buttonPress == '#') {
    Serial.println("Emergency alert key pressed");
    emergencyAlert();
  }

  if (buttonPress == '*') {
    Serial.println("Warning alert key pressed");
    warningAlert();
  }
}
